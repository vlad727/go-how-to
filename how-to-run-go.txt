func main ### имя main является особенным, эта функция будет вызываться сама при запуске программы 
fmt.Println("Hello World") ### доступ к функции пакета fmt под названием Println (Print line), затем создание новой строки, содержащей Hello World, и вызов функции с этой строкой в качестве первого и единственного аргумента.
